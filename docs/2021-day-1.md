data = """

result = (data  
 | split['\n']  
 | trim['']                   # clean up artifacts  
 | map[int]  
 | sliding[2]                 # all consecutive pairs  
 | filter[unpack[less_than]]   
 | length                     # how many?  
 | collect                    # trigger evaluation  
)
